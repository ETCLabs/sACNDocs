236 extern "C" inline void SourceDetectorCbSourceUpdated(sacn_source_detector_t handle, const EtcPalUuid* cid,
237  const char* name, const uint16_t* sourced_universes,
238  size_t num_sourced_universes, void* context)
239 {
240  if (context && cid && name)
241  {
242  std::vector<uint16_t> sourced_vec;
243  if (sourced_universes && (num_sourced_universes > 0))
244  sourced_vec.assign(sourced_universes, sourced_universes + num_sourced_universes);
245  static_cast<SourceDetector::NotifyHandler*>(context)->HandleSourceUpdated(handle, *cid, name, sourced_vec);
246  }
247 }

450 inline std::vector<SacnMcastInterface> SourceDetector::GetNetworkInterfaces()
451 {
452  // This uses a guessing algorithm with a while loop to avoid race conditions.
453  std::vector<SacnMcastInterface> netints;
454  size_t size_guess = 4u;
455  size_t num_netints = 0u;
456 
457  do
458  {
459  netints.resize(size_guess);
460  num_netints = sacn_source_detector_get_network_interfaces(handle_, netints.data(), netints.size());
461  size_guess = num_netints + 4u;
462  } while (num_netints > netints.size());
463 
464  netints.resize(num_netints);
465  return netints;
466 }
