220 extern "C" inline void SourceDetectorCbSourceUpdated(const EtcPalUuid* cid, const char* name,
221  const uint16_t* sourced_universes, size_t num_sourced_universes,
222  void* context)
223 {
224  if (context && cid && name)
225  {
226  std::vector<uint16_t> sourced_vec;
227  if (sourced_universes && (num_sourced_universes > 0))
228  sourced_vec.assign(sourced_universes, sourced_universes + num_sourced_universes);
229  static_cast<SourceDetector::NotifyHandler*>(context)->HandleSourceUpdated(*cid, name, sourced_vec);
230  }
231 }

430 inline std::vector<SacnMcastInterface> SourceDetector::GetNetworkInterfaces()
431 {
432  // This uses a guessing algorithm with a while loop to avoid race conditions.
433  std::vector<SacnMcastInterface> netints;
434  size_t size_guess = 4u;
435  size_t num_netints = 0u;
436 
437  do
438  {
439  netints.resize(size_guess);
440  num_netints = sacn_source_detector_get_network_interfaces(netints.data(), netints.size());
441  size_guess = num_netints + 4u;
442  } while (num_netints > netints.size());
443 
444  netints.resize(num_netints);
445  return netints;
446 }
