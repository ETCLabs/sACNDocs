388 inline std::vector<EtcPalMcastNetintId> MergeReceiver::GetNetworkInterfaces()
389 {
390  // This uses a guessing algorithm with a while loop to avoid race conditions.
391  std::vector<EtcPalMcastNetintId> netints;
392  size_t size_guess = 4u;
393  size_t num_netints = 0u;
394 
395  do
396  {
397  netints.resize(size_guess);
398  num_netints = sacn_merge_receiver_get_network_interfaces(handle_, netints.data(), netints.size());
399  size_guess = num_netints + 4u;
400  } while (num_netints > netints.size());
401 
402  netints.resize(num_netints);
403  return netints;
404 }

495 inline etcpal::Error MergeReceiver::ResetNetworking(std::vector<SacnMcastInterface>& sys_netints,
496  std::vector<NetintList>& per_receiver_netint_lists)
497 {
498  std::vector<SacnMergeReceiverNetintList> netint_lists_c;
499  netint_lists_c.reserve(per_receiver_netint_lists.size());
500  std::transform(per_receiver_netint_lists.begin(), per_receiver_netint_lists.end(), std::back_inserter(netint_lists_c),
501  [](NetintList& list) {
502  // clang-format off
503  SacnMergeReceiverNetintList c_list = {
504  list.handle,
505  list.netints.data(),
506  list.netints.size()
507  };
508  // clang-format on
509 
510  return c_list;
511  });
512 
513  SacnNetintConfig netint_config = {sys_netints.data(), sys_netints.size()};
514  return sacn_merge_receiver_reset_networking_per_receiver(&netint_config, netint_lists_c.data(),
