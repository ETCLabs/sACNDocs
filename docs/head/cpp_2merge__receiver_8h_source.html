443 inline std::vector<EtcPalMcastNetintId> MergeReceiver::GetNetworkInterfaces()
444 {
445  // This uses a guessing algorithm with a while loop to avoid race conditions.
446  std::vector<EtcPalMcastNetintId> netints;
447  size_t size_guess = 4u;
448  size_t num_netints = 0u;
449 
450  do
451  {
452  netints.resize(size_guess);
453  num_netints = sacn_merge_receiver_get_network_interfaces(handle_, netints.data(), netints.size());
454  size_guess = num_netints + 4u;
455  } while (num_netints > netints.size());
456 
457  netints.resize(num_netints);
458  return netints;
459 }

550 inline etcpal::Error MergeReceiver::ResetNetworking(std::vector<SacnMcastInterface>& sys_netints,
551  std::vector<NetintList>& per_receiver_netint_lists)
552 {
553  std::vector<SacnMergeReceiverNetintList> netint_lists_c;
554  netint_lists_c.reserve(per_receiver_netint_lists.size());
555  std::transform(per_receiver_netint_lists.begin(), per_receiver_netint_lists.end(), std::back_inserter(netint_lists_c),
556  [](NetintList& list) {
557  // clang-format off
558  SacnMergeReceiverNetintList c_list = {
559  list.handle,
560  list.netints.data(),
561  list.netints.size()
562  };
563  // clang-format on
564 
565  return c_list;
566  });
567 
568  SacnNetintConfig netint_config = {sys_netints.data(), sys_netints.size()};
569  return sacn_merge_receiver_reset_networking_per_receiver(&netint_config, netint_lists_c.data(),
