393 inline std::vector<EtcPalMcastNetintId> MergeReceiver::GetNetworkInterfaces()
394 {
395  // This uses a guessing algorithm with a while loop to avoid race conditions.
396  std::vector<EtcPalMcastNetintId> netints;
397  size_t size_guess = 4u;
398  size_t num_netints = 0u;
399 
400  do
401  {
402  netints.resize(size_guess);
403  num_netints = sacn_merge_receiver_get_network_interfaces(handle_, netints.data(), netints.size());
404  size_guess = num_netints + 4u;
405  } while (num_netints > netints.size());
406 
407  netints.resize(num_netints);
408  return netints;
409 }

500 inline etcpal::Error MergeReceiver::ResetNetworking(std::vector<SacnMcastInterface>& sys_netints,
501  std::vector<NetintList>& per_receiver_netint_lists)
502 {
503  std::vector<SacnMergeReceiverNetintList> netint_lists_c;

506  [](NetintList& list) {

519  return sacn_merge_receiver_reset_networking_per_receiver(&netint_config, netint_lists_c.data(),
