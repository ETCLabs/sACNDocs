424 inline std::vector<uint16_t> Source::GetUniverses()
425 {
426  // This uses a guessing algorithm with a while loop to avoid race conditions.
427  std::vector<uint16_t> universes;
428  size_t size_guess = 4u;
429  size_t num_universes = 0u;
430 
431  do
432  {
433  universes.resize(size_guess);
434  num_universes = sacn_source_get_universes(handle_.value(), universes.data(), universes.size());
435  size_guess = num_universes + 4u;
436  } while (num_universes > universes.size());
437 
438  universes.resize(num_universes);
439  return universes;
440 }

482 inline std::vector<etcpal::IpAddr> Source::GetUnicastDestinations(uint16_t universe)
483 {
484  // This uses a guessing algorithm with a while loop to avoid race conditions.
485  std::vector<EtcPalIpAddr> destinations;
486  size_t size_guess = 4u;
487  size_t num_destinations = 0u;
488 
489  do
490  {
491  destinations.resize(size_guess);
492  num_destinations =
493  sacn_source_get_unicast_destinations(handle_.value(), universe, destinations.data(), destinations.size());
494  size_guess = num_destinations + 4u;
495  } while (num_destinations > destinations.size());
496 
497  destinations.resize(num_destinations);
498 
499  // Convert vector<EtcPalIpAddr> to vector<etcpal::IpAddr>.
500  std::vector<etcpal::IpAddr> result;
501  if (!destinations.empty())
502  {
503  result.reserve(destinations.size());
504  std::transform(destinations.begin(), destinations.end(), std::back_inserter(result),
505  [](const EtcPalIpAddr& dest) { return etcpal::IpAddr(dest); });
506  }
507 
508  return result;
509 }
