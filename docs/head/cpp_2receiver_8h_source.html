440 inline std::vector<EtcPalMcastNetintId> Receiver::GetNetworkInterfaces()
441 {
442  // This uses a guessing algorithm with a while loop to avoid race conditions.
443  std::vector<EtcPalMcastNetintId> netints;
444  size_t size_guess = 4u;
445  size_t num_netints = 0u;
446 
447  do
448  {
449  netints.resize(size_guess);
450  num_netints = sacn_receiver_get_network_interfaces(handle_, netints.data(), netints.size());
451  size_guess = num_netints + 4u;
452  } while (num_netints > netints.size());
453 
454  netints.resize(num_netints);
455  return netints;
456 }
