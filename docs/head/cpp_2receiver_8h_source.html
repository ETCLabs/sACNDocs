503 inline std::vector<EtcPalMcastNetintId> Receiver::GetNetworkInterfaces()
504 {
505  // This uses a guessing algorithm with a while loop to avoid race conditions.
506  std::vector<EtcPalMcastNetintId> netints;
507  size_t size_guess = 4u;
508  size_t num_netints = 0u;
509 
510  do
511  {
512  netints.resize(size_guess);
513  num_netints = sacn_receiver_get_network_interfaces(handle_, netints.data(), netints.size());
514  size_guess = num_netints + 4u;
515  } while (num_netints > netints.size());
516 
517  netints.resize(num_netints);
518  return netints;
519 }
