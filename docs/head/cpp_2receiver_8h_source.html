444 inline std::vector<EtcPalMcastNetintId> Receiver::GetNetworkInterfaces()
445 {
446  // This uses a guessing algorithm with a while loop to avoid race conditions.
447  std::vector<EtcPalMcastNetintId> netints;
448  size_t size_guess = 4u;
449  size_t num_netints = 0u;
450 
451  do
452  {
453  netints.resize(size_guess);
454  num_netints = sacn_receiver_get_network_interfaces(handle_, netints.data(), netints.size());
455  size_guess = num_netints + 4u;
456  } while (num_netints > netints.size());
457 
458  netints.resize(num_netints);
459  return netints;
460 }
