424 inline etcpal::Error MergeReceiver::Startup(const Settings& settings, NotifyHandler& notify_handler,
425  std::vector<SacnMcastInterface>& netints)
426 {
427  SacnMergeReceiverConfig config = TranslateConfig(settings, notify_handler);
428 
429  sacn_merge_receiver_t c_handle = SACN_MERGE_RECEIVER_INVALID;
430  etcpal::Error result = kEtcPalErrOk;
431 
432  if (netints.empty())
433  {
434  result = sacn_merge_receiver_create(&config, &c_handle, NULL);
435  }
436  else
437  {
438  SacnNetintConfig netint_config = {netints.data(), netints.size()};
439  result = sacn_merge_receiver_create(&config, &c_handle, &netint_config);
440  }
441 
442  handle_.SetValue(c_handle);
443 
444  return result;
445 }

546 inline std::vector<EtcPalMcastNetintId> MergeReceiver::GetNetworkInterfaces()
547 {
548  // This uses a guessing algorithm with a while loop to avoid race conditions.
549  std::vector<EtcPalMcastNetintId> netints;
550  size_t size_guess = 4u;
551  size_t num_netints = 0u;
552 
553  do
554  {
555  netints.resize(size_guess);
556  num_netints = sacn_merge_receiver_get_network_interfaces(handle_.value(), netints.data(), netints.size());
557  size_guess = num_netints + 4u;
558  } while (num_netints > netints.size());
559 
560  netints.resize(num_netints);
561  return netints;
562 }

680 inline etcpal::Error MergeReceiver::ResetNetworking(std::vector<SacnMcastInterface>& sys_netints,
681  std::vector<NetintList>& per_receiver_netint_lists)
682 {

685  std::transform(per_receiver_netint_lists.begin(), per_receiver_netint_lists.end(), std::back_inserter(netint_lists_c),

698  SacnNetintConfig netint_config = {sys_netints.data(), sys_netints.size()};
