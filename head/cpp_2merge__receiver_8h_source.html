453inline bool MergeReceiver::Settings::IsValid() const
454{
455 return (universe_id > 0) && (footprint.start_address >= 1) &&
456 (footprint.start_address <= SACN_DMX_MERGER_MAX_SLOTS) && (footprint.address_count >= 1) &&
457 (footprint.address_count <= (SACN_DMX_MERGER_MAX_SLOTS - footprint.start_address + 1));

508inline etcpal::Error MergeReceiver::Startup(const Settings& settings,
509 NotifyHandler& notify_handler,
510 McastMode mcast_mode = McastMode::kEnabledOnAllInterfaces)
511{
512 SacnMergeReceiverConfig config = TranslateConfig(settings, notify_handler);
513
514 SacnNetintConfig netint_config = SACN_NETINT_CONFIG_DEFAULT_INIT;
515 if (mcast_mode == McastMode::kDisabledOnAllInterfaces)
516 netint_config.no_netints = true;
517
518 sacn_merge_receiver_t c_handle = kSacnMergeReceiverInvalid;
519 etcpal::Error result = sacn_merge_receiver_create(&config, &c_handle, &netint_config);
520
521 handle_.SetValue(c_handle);
522
523 return result;
524}

549inline etcpal::Error MergeReceiver::Startup(const Settings& settings,
550 NotifyHandler& notify_handler,
551 std::vector<SacnMcastInterface>& netints)
552{
553 SacnMergeReceiverConfig config = TranslateConfig(settings, notify_handler);
554
555 sacn_merge_receiver_t c_handle = kSacnMergeReceiverInvalid;
556 etcpal::Error result = kEtcPalErrOk;
557
558 if (netints.empty())
559 {
560 result = sacn_merge_receiver_create(&config, &c_handle, nullptr);
561 }
562 else
563 {
564 SacnNetintConfig netint_config = SACN_NETINT_CONFIG_DEFAULT_INIT;
565 netint_config.netints = netints.data();
566 netint_config.num_netints = netints.size();
567
568 result = sacn_merge_receiver_create(&config, &c_handle, &netint_config);
569 }
570
571 handle_.SetValue(c_handle);
572
573 return result;
574}

679inline std::vector<EtcPalMcastNetintId> MergeReceiver::GetNetworkInterfaces()
680{
681 // This uses a guessing algorithm with a while loop to avoid race conditions.
682 size_t size_guess = 4u;
683 std::vector<EtcPalMcastNetintId> netints(size_guess);
684 size_t num_netints = sacn_merge_receiver_get_network_interfaces(handle_.value(), netints.data(), netints.size());
685
686 while (num_netints > netints.size())
687 {
688 size_guess = num_netints + 4u;
689 netints.resize(size_guess);
690 num_netints = sacn_merge_receiver_get_network_interfaces(handle_.value(), netints.data(), netints.size());
691 }
692
693 netints.resize(num_netints);
694 return netints;
695}

707inline etcpal::Expected<MergeReceiver::Source> MergeReceiver::GetSource(sacn_remote_source_t source_handle)
708{
709 SacnMergeReceiverSource c_info;
710 etcpal_error_t error = sacn_merge_receiver_get_source(handle_.value(), source_handle, &c_info);
711 if (error == kEtcPalErrOk)
712 {
713 MergeReceiver::Source res;
714 res.handle = c_info.handle;
715 res.cid = c_info.cid;
716 res.name = c_info.name;
717 res.addr = c_info.addr;
718 res.per_address_priorities_active = c_info.per_address_priorities_active;
719 res.universe_priority = c_info.universe_priority;
720 return res;
721 }
722
723 return error;
724}

823inline etcpal::Error MergeReceiver::ResetNetworking(std::vector<SacnMcastInterface>& sys_netints,
824 std::vector<NetintList>& per_receiver_netint_lists)
825{
826 std::vector<SacnMergeReceiverNetintList> netint_lists_c;
827 netint_lists_c.reserve(per_receiver_netint_lists.size());
828 std::transform(per_receiver_netint_lists.begin(), per_receiver_netint_lists.end(), std::back_inserter(netint_lists_c),
829 [](NetintList& list) {
830 // clang-format off
831 SacnMergeReceiverNetintList c_list = {
832 list.handle,
833 list.netints.data(),
834 list.netints.size(),
835 list.no_netints
836 };
837 // clang-format on
838
839 return c_list;
840 });
841
842 SacnNetintConfig netint_config = SACN_NETINT_CONFIG_DEFAULT_INIT;
843 netint_config.netints = sys_netints.data();
844 netint_config.num_netints = sys_netints.size();
845
846 return sacn_merge_receiver_reset_networking_per_receiver(&netint_config, netint_lists_c.data(),
847 netint_lists_c.size());
848}
