433inline etcpal::Error Receiver::Startup(const Settings& settings,
434 NotifyHandler& notify_handler,
435 McastMode mcast_mode = McastMode::kEnabledOnAllInterfaces)
436{
437 SacnReceiverConfig config = TranslateConfig(settings, notify_handler);
438
439 SacnNetintConfig netint_config = SACN_NETINT_CONFIG_DEFAULT_INIT;
440 if (mcast_mode == McastMode::kDisabledOnAllInterfaces)
441 netint_config.no_netints = true;
442
443 sacn_receiver_t c_handle = kSacnReceiverInvalid;
444 etcpal::Error result = sacn_receiver_create(&config, &c_handle, &netint_config);
445
446 handle_.SetValue(c_handle);
447
448 return result;
449}

478inline etcpal::Error Receiver::Startup(const Settings& settings,
479 NotifyHandler& notify_handler,
480 std::vector<SacnMcastInterface>& netints)
481{
482 SacnReceiverConfig config = TranslateConfig(settings, notify_handler);
483
484 sacn_receiver_t c_handle = kSacnReceiverInvalid;
485 etcpal::Error result = kEtcPalErrOk;
486
487 if (netints.empty())
488 {
489 result = sacn_receiver_create(&config, &c_handle, nullptr);
490 }
491 else
492 {
493 SacnNetintConfig netint_config = SACN_NETINT_CONFIG_DEFAULT_INIT;
494 netint_config.netints = netints.data();
495 netint_config.num_netints = netints.size();
496
497 result = sacn_receiver_create(&config, &c_handle, &netint_config);
498 }
499
500 handle_.SetValue(c_handle);
501
502 return result;
503}

612inline std::vector<EtcPalMcastNetintId> Receiver::GetNetworkInterfaces()
613{
614 // This uses a guessing algorithm with a while loop to avoid race conditions.
615 size_t size_guess = 4u;
616 std::vector<EtcPalMcastNetintId> netints(size_guess);
617 size_t num_netints = sacn_receiver_get_network_interfaces(handle_.value(), netints.data(), netints.size());
618
619 while (num_netints > netints.size())
620 {
621 size_guess = num_netints + 4u;
622 netints.resize(size_guess);
623 num_netints = sacn_receiver_get_network_interfaces(handle_.value(), netints.data(), netints.size());
624 }
625
626 netints.resize(num_netints);
627 return netints;
628}

754inline etcpal::Error Receiver::ResetNetworking(std::vector<SacnMcastInterface>& sys_netints,
755 std::vector<NetintList>& per_receiver_netint_lists)
756{
757 std::vector<SacnReceiverNetintList> netint_lists_c;
758 netint_lists_c.reserve(per_receiver_netint_lists.size());
759 std::transform(per_receiver_netint_lists.begin(), per_receiver_netint_lists.end(), std::back_inserter(netint_lists_c),
760 [](NetintList& list) {
761 // clang-format off
762 SacnReceiverNetintList c_list = {
763 list.handle,
764 list.netints.data(),
765 list.netints.size(),
766 list.no_netints
767 };
768 // clang-format on
769
770 return c_list;
771 });
772
773 SacnNetintConfig sys_netint_config = SACN_NETINT_CONFIG_DEFAULT_INIT;
774 sys_netint_config.netints = sys_netints.data();
775 sys_netint_config.num_netints = sys_netints.size();
776
777 return sacn_receiver_reset_networking_per_receiver(&sys_netint_config, netint_lists_c.data(), netint_lists_c.size());
778}
