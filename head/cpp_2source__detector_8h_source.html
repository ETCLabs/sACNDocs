259extern "C" inline void SourceDetectorCbSourceUpdated(sacn_remote_source_t handle,
260 const EtcPalUuid* cid,
261 const char* name,
262 const uint16_t* sourced_universes,
263 size_t num_sourced_universes,
264 void* context)
265{
266 if (context && cid && name)
267 {
268 std::vector<uint16_t> sourced_vec;
269 if (sourced_universes && (num_sourced_universes > 0))
270 sourced_vec.assign(sourced_universes, sourced_universes + num_sourced_universes);
271 static_cast<SourceDetector::NotifyHandler*>(context)->HandleSourceUpdated(handle, *cid, name, sourced_vec);
272 }
273}
274
275extern "C" inline void SourceDetectorCbSourceExpired(sacn_remote_source_t handle,
276 const EtcPalUuid* cid,
277 const char* name,
278 void* context)
279{
280 if (context && cid && name)
281 {
282 static_cast<SourceDetector::NotifyHandler*>(context)->HandleSourceExpired(handle, *cid, name);

284}

321{
322 SacnSourceDetectorConfig config = TranslateConfig(Settings(), notify_handler);

512inline std::vector<EtcPalMcastNetintId> SourceDetector::GetNetworkInterfaces()
513{
514 // This uses a guessing algorithm with a while loop to avoid race conditions.
515 size_t size_guess = 4u;
516 std::vector<EtcPalMcastNetintId> netints(size_guess);
517 size_t num_netints = sacn_source_detector_get_network_interfaces(netints.data(), netints.size());
518
519 while (num_netints > netints.size())
520 {
521 size_guess = num_netints + 4u;
522 netints.resize(size_guess);
523 num_netints = sacn_source_detector_get_network_interfaces(netints.data(), netints.size());
524 }
525
526 netints.resize(num_netints);
527 return netints;
528}
