445 inline std::vector<uint16_t> Source::GetUniverses()
446 {
447  // This uses a guessing algorithm with a while loop to avoid race conditions.
448  std::vector<uint16_t> universes;
449  size_t size_guess = 4u;
450  size_t num_universes = 0u;
451 
452  do
453  {
454  universes.resize(size_guess);
455  num_universes = sacn_source_get_universes(handle_.value(), universes.data(), universes.size());
456  size_guess = num_universes + 4u;
457  } while (num_universes > universes.size());
458 
459  universes.resize(num_universes);
460  return universes;
461 }

503 inline std::vector<etcpal::IpAddr> Source::GetUnicastDestinations(uint16_t universe)
504 {
505  // This uses a guessing algorithm with a while loop to avoid race conditions.
506  std::vector<EtcPalIpAddr> destinations;
507  size_t size_guess = 4u;
508  size_t num_destinations = 0u;
509 
510  do
511  {
512  destinations.resize(size_guess);
513  num_destinations =
514  sacn_source_get_unicast_destinations(handle_.value(), universe, destinations.data(), destinations.size());
515  size_guess = num_destinations + 4u;
516  } while (num_destinations > destinations.size());
517 
518  destinations.resize(num_destinations);
519 
520  // Convert vector<EtcPalIpAddr> to vector<etcpal::IpAddr>.
521  std::vector<etcpal::IpAddr> result;
522  if (!destinations.empty())
523  {
524  result.reserve(destinations.size());
525  std::transform(destinations.begin(), destinations.end(), std::back_inserter(result),
526  [](const EtcPalIpAddr& dest) { return etcpal::IpAddr(dest); });
527  }
528 
529  return result;
530 }
