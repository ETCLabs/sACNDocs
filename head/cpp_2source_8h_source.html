471inline std::vector<uint16_t> Source::GetUniverses()
472{
473 // This uses a guessing algorithm with a while loop to avoid race conditions.
474 size_t size_guess = 4u;
475 std::vector<uint16_t> universes(size_guess);
476 size_t num_universes = sacn_source_get_universes(handle_.value(), universes.data(), universes.size());
477
478 while (num_universes > universes.size())
479 {
480 size_guess = num_universes + 4u;
481 universes.resize(size_guess);
482 num_universes = sacn_source_get_universes(handle_.value(), universes.data(), universes.size());
483 }
484
485 universes.resize(num_universes);
486 return universes;
487}

529inline std::vector<etcpal::IpAddr> Source::GetUnicastDestinations(uint16_t universe)
530{
531 // This uses a guessing algorithm with a while loop to avoid race conditions.
532 size_t size_guess = 4u;
533 std::vector<EtcPalIpAddr> destinations(size_guess);
534 size_t num_destinations =
535 sacn_source_get_unicast_destinations(handle_.value(), universe, destinations.data(), destinations.size());
536
537 while (num_destinations > destinations.size())
538 {
539 size_guess = num_destinations + 4u;
540 destinations.resize(size_guess);
541 num_destinations =
542 sacn_source_get_unicast_destinations(handle_.value(), universe, destinations.data(), destinations.size());
543 }
544
545 destinations.resize(num_destinations);
546
547 // Convert vector<EtcPalIpAddr> to vector<etcpal::IpAddr>.
548 std::vector<etcpal::IpAddr> result;
549 if (!destinations.empty())
550 {
551 result.reserve(destinations.size());
552 std::transform(destinations.begin(), destinations.end(), std::back_inserter(result),
553 [](const EtcPalIpAddr& dest) { return etcpal::IpAddr(dest); });
554 }
555
556 return result;
557}
