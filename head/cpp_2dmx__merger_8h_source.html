187inline etcpal::Error DmxMerger::Startup(const Settings& settings)
188{
189 SacnDmxMergerConfig config = TranslateConfig(settings);
190
191 sacn_dmx_merger_t c_handle = kSacnDmxMergerInvalid;
192 etcpal::Error result = sacn_dmx_merger_create(&config, &c_handle);
193
194 handle_.SetValue(c_handle);
195
196 return result;
197}

204inline void DmxMerger::Shutdown()

206 sacn_dmx_merger_destroy(handle_.value());
207 handle_.Clear();
208}
