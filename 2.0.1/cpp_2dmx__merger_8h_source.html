184inline etcpal::Error DmxMerger::Startup(const Settings& settings)
185{
186 SacnDmxMergerConfig config = TranslateConfig(settings);
187
188 sacn_dmx_merger_t c_handle = SACN_DMX_MERGER_INVALID;
189 etcpal::Error result = sacn_dmx_merger_create(&config, &c_handle);
190
191 handle_.SetValue(c_handle);
192
193 return result;
194}
